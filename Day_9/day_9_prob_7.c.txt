#include<stdio.h>
int main()
{
    int ADC0NO=0x20,CMCON=0,mask;

    ADC0NO=(ADC0NO|mask);

    {
        CMCON=((CMCON)|((1<<7)|(1<<6)|(1<<3)));
        printf("the value of CMCON is %x",CMCON);
    }

}
